Minería de datos con Matminer#

from matminer.datasets import get_available_datasets, load_dataset
import io, contextlib, warnings

warnings.filterwarnings("ignore")  

CANDIDATE_COLS = [
    "formula", "formula_pretty", "pretty_formula",
    "chemical_formula", "composition"
]
TARGET = ["Ti", "C","O"]

# obtener lista de datasets 
buf = io.StringIO()
with contextlib.redirect_stdout(buf):
    datasets = get_available_datasets()

if isinstance(datasets, dict):
    dataset_names = list(datasets.keys())
else:
    dataset_names = [d if isinstance(d, str) else d[0] for d in datasets]

for ds_name in dataset_names:
    try:
        # cargar dataset
        buf = io.StringIO()
        with contextlib.redirect_stdout(buf):
            df = load_dataset(ds_name, download_if_missing=False)
    except Exception:
        continue

    found_col = next((c for c in CANDIDATE_COLS if c in df.columns), None)
    if not found_col:
        continue

    for idx, val in df[found_col].dropna().items():
        val_str = str(val)
        if all(el in val_str for el in TARGET):
            print(f"{ds_name}\tindex={idx}\tcol={found_col}\tformula={val_str}")
